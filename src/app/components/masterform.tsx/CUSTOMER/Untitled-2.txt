"use client";

import React, { useEffect, useState } from "react";
import {
  User,
  Mail,
  Phone,
  Calendar,
  AlertCircle,
  CheckCircle2,
  Users,
  MapPin,
  Home,
} from "lucide-react";
import {
  getAllCitiesApi,
  getStatesByCityApi,
} from "../../../features/State/stateApi";
import { createCustomerApi } from "../../../features/Customer/customerApi";

interface CustomerRecord {
  firstName: string;
  lastName: string;
  email: string;
  phone: string;
  dateOfBirth: string;
  anniversary: string;
  gender: string;
  address: string;
  state: string;
  city: string;
  pincode: string;
  stateId?: number;
  cityId?: number;
}

interface City {
  id: number;
  cityName: string;
}
interface State {
  id: number;
  stateName: string;
}

const initialFormState: CustomerRecord = {
  firstName: "",
  lastName: "",
  email: "",
  phone: "",
  dateOfBirth: "",
  anniversary: "",
  gender: "",
  address: "",
  state: "",
  city: "",
  pincode: "",
};

const CustomerForm = () => {
  const [customerFormData, setFormData] =
    useState<CustomerRecord>(initialFormState);
  const [touchedFields, setTouchedFields] = useState<Record<string, boolean>>(
    {},
  );
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [isSuccess, setIsSuccess] = useState(false);

  const [allCities, setAllCities] = useState<City[]>([]);
  const [availableStates, setAvailableStates] = useState<State[]>([]);
  const [isLoadingCities, setIsLoadingCities] = useState(true);
  const [isLoadingStates, setIsLoadingStates] = useState(false);

  const genderOptions = ["Male", "Female", "Other"];

  // ─── Cities load on mount ───────────────────────────────────────────────
  useEffect(() => {
    const loadCities = async () => {
      try {
        setIsLoadingCities(true);
        const cities = await getAllCitiesApi();
        // Unique city names
        const unique = [
          ...new Map(cities.map((c) => [c.cityName, c])).values(),
        ];
        setAllCities(unique);
      } catch (err) {
        console.error("Cities fetch error:", err);
      } finally {
        setIsLoadingCities(false);
      }
    };
    loadCities();
  }, []);

  // ─── City change → states auto-load ────────────────────────────────────
  const handleCityChange = async (e: React.ChangeEvent<HTMLSelectElement>) => {
    const selectedCity = e.target.value;

    // City set karo, state reset karo
    setFormData((prev) => ({
      ...prev,
      city: selectedCity,
      state: "",
      stateId: undefined,
    }));

    if (errors.city) setErrors((prev) => ({ ...prev, city: "" }));
    setAvailableStates([]);

    if (!selectedCity) return;

    try {
      setIsLoadingStates(true);
      const states = await getStatesByCityApi(selectedCity);

      setAvailableStates(states);

      if (states.length === 1) {
        setFormData((prev) => ({
          ...prev,
          state: states[0].stateName,
          stateId: states[0].id,
        }));
      }
    } catch (err) {
      console.error("States fetch error:", err);
    } finally {
      setIsLoadingStates(false);
    }
  };

  const handleStateChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    const selectedStateName = e.target.value;
    const selectedState = availableStates.find(
      (s) => s.stateName === selectedStateName,
    );

    setFormData((prev) => ({
      ...prev,
      state: selectedStateName,
      stateId: selectedState?.id,
    }));

    if (errors.state) setErrors((prev) => ({ ...prev, state: "" }));
  };

  const handleFieldChange = (
    e: React.ChangeEvent<
      HTMLInputElement | HTMLSelectElement | HTMLTextAreaElement
    >,
  ) => {
    const { name, value } = e.target;
    setFormData((prev) => ({ ...prev, [name]: value }));
    if (errors[name]) setErrors((prev) => ({ ...prev, [name]: "" }));
  };

  const markFieldTouched = (fieldName: string) => {
    setTouchedFields((prev) => ({ ...prev, [fieldName]: true }));
  };

  const validateForm = (): boolean => {
    const newErrors: Record<string, string> = {};
    const requiredFields: Array<keyof CustomerRecord> = [
      "firstName",
      "lastName",
      "email",
      "phone",
      "address",
      "state",
      "city",
      "pincode",
    ];

    requiredFields.forEach((field) => {
      if (!customerFormData[field]?.trim()) {
        const label =
          field === "firstName"
            ? "First name"
            : field === "lastName"
              ? "Last name"
              : field.charAt(0).toUpperCase() + field.slice(1);
        newErrors[field] = `${label} is required`;
      }
    });

    if (customerFormData.email && !/\S+@\S+\.\S+/.test(customerFormData.email))
      newErrors.email = "Enter valid email";

    if (customerFormData.phone && !/^\d{10}$/.test(customerFormData.phone))
      newErrors.phone = "Phone must be 10 digits";

    if (customerFormData.pincode && !/^\d{6}$/.test(customerFormData.pincode))
      newErrors.pincode = "Pincode must be 6 digits";

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  // ─── Submit ─────────────────────────────────────────────────────────────
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    const allTouched: Record<string, boolean> = {};
    Object.keys(customerFormData).forEach((k) => (allTouched[k] = true));
    setTouchedFields(allTouched);

    if (!validateForm()) return;

    try {
      setIsLoading(true);
      setError(null);
      await createCustomerApi(customerFormData);
      setIsSuccess(true);
      setFormData(initialFormState);
      setAvailableStates([]);
      window.scrollTo({ top: 0, behavior: "smooth" });
      setTimeout(() => setIsSuccess(false), 3000);
    } catch (err: any) {
      setError(err?.message || "Registration failed");
    } finally {
      setIsLoading(false);
    }
  };

  const getInputClass = (field: string, icon = true) => {
    const base = `w-full ${icon ? "pl-10" : "px-4"} pr-4 py-3 border rounded-lg focus:outline-none focus:ring-2 transition-colors`;
    if (touchedFields[field] && errors[field])
      return `${base} border-red-500 bg-red-50 focus:ring-red-200`;
    return `${base} border-gray-300 focus:ring-blue-200 focus:border-blue-400`;
  };

  return (
    <div className="w-full bg-gray-50 min-h-screen">
      {/* Header */}
      <div className="sticky top-0 z-30 bg-orange-100 p-3 rounded-md shadow-sm">
        <div className="flex justify-between items-center">
          <div className="pl-4 border-l-8 border-orange-500 bg-white px-3 rounded-md shadow-md">
            <h2 className="text-4xl font-bold text-left py-4 text-orange-600">
              Customer Registration Form
            </h2>
          </div>
        </div>
      </div>

      <div className="p-6 w-full mx-auto bg-white shadow-xl rounded-lg my-6">
        {/* Status Messages */}
        {error && (
          <div className="flex items-center gap-2 p-4 mb-6 text-red-700 bg-red-50 rounded-lg border border-red-200">
            <AlertCircle size={20} />
            <span className="font-medium">{error}</span>
          </div>
        )}
        {isSuccess && (
          <div className="fixed top-6 right-6 z-50 flex items-center gap-2 px-5 py-3 text-green-700 bg-green-50 border border-green-300 rounded-lg shadow-lg">
            <CheckCircle2 size={20} />
            <span className="font-medium">
              Customer registered successfully!
            </span>
          </div>
        )}

        <form onSubmit={handleSubmit} className="space-y-6">
          {/* Personal Information */}
          <div className="p-6 border rounded-xl bg-purple-50">
            <h3 className="text-xl font-semibold text-blue-800 mb-6 pb-3 border-b">
              <span className="bg-blue-600 text-white px-3 py-1 rounded-md mr-2">
                1
              </span>
              Personal Information
            </h3>
            <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
              <Input
                label="First Name"
                name="firstName"
                icon={<User size={18} />}
                value={customerFormData.firstName}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.firstName}
                touched={touchedFields.firstName}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Last Name"
                name="lastName"
                icon={<User size={18} />}
                value={customerFormData.lastName}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.lastName}
                touched={touchedFields.lastName}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Email"
                name="email"
                type="email"
                icon={<Mail size={18} />}
                value={customerFormData.email}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.email}
                touched={touchedFields.email}
                getInputClass={getInputClass}
                required
              />

              <Input
                label="Phone"
                name="phone"
                type="tel"
                icon={<Phone size={18} />}
                maxLength={10}
                value={customerFormData.phone}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.phone}
                touched={touchedFields.phone}
                getInputClass={getInputClass}
                required
                placeholder="10-digit mobile number"
              />

              <Input
                label="Date of Birth"
                name="dateOfBirth"
                type="date"
                icon={<Calendar size={18} />}
                value={customerFormData.dateOfBirth}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.dateOfBirth}
                touched={touchedFields.dateOfBirth}
                getInputClass={getInputClass}
              />

              <Input
                label="Anniversary"
                name="anniversary"
                type="date"
                icon={<Calendar size={18} />}
                value={customerFormData.anniversary}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                error={errors.anniversary}
                touched={touchedFields.anniversary}
                getInputClass={getInputClass}
              />

              <SelectField
                label="Gender"
                name="gender"
                icon={<Users size={18} />}
                options={genderOptions}
                value={customerFormData.gender}
                onChange={handleFieldChange}
                onBlur={markFieldTouched}
                getInputClass={getInputClass}
              />
            </div>
          </div>

          {/* Address Information */}
          <div className="p-6 border rounded-xl bg-green-50">
            <h3 className="mb-4 text-xl font-semibold text-green-800 flex items-center gap-2">
              <Home size={20} /> Address Information
            </h3>

            <div className="mb-4">
              <label className="block font-medium text-gray-700 mb-1">
                Address <span className="text-red-500">*</span>
              </label>
              <textarea
                name="address"
                placeholder="Enter complete address"
                value={customerFormData.address}
                onChange={handleFieldChange}
                onBlur={() => markFieldTouched("address")}
                className={getInputClass("address", false)}
                rows={3}
              />
              {touchedFields.address && errors.address && (
                <p className="text-sm text-red-600 mt-1">{errors.address}</p>
              )}
            </div>

            <div className="grid gap-4 md:grid-cols-3">
              {/* City Dropdown */}
              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  City <span className="text-red-500">*</span>
                </label>
                <div className="relative">
                  <MapPin
                    className="absolute left-3 top-3 text-gray-400 z-10"
                    size={18}
                  />
                  <select
                    name="city"
                    value={customerFormData.city}
                    onChange={handleCityChange}
                    onBlur={() => markFieldTouched("city")}
                    className={getInputClass("city")}
                    disabled={isLoadingCities}
                  >
                    <option value="">
                      {isLoadingCities ? "Loading cities..." : "Select City"}
                    </option>
                    {allCities.map((city) => (
                      <option key={city.id} value={city.cityName}>
                        {city.cityName}
                      </option>
                    ))}
                  </select>
                </div>
                {touchedFields.city && errors.city && (
                  <p className="text-sm text-red-600 mt-1">{errors.city}</p>
                )}
              </div>

              {/* State Dropdown — city select hone ke baad auto-populate */}
              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  State <span className="text-red-500">*</span>
                </label>
                <div className="relative">
                  <MapPin
                    className="absolute left-3 top-3 text-gray-400 z-10"
                    size={18}
                  />
                  <select
                    name="state"
                    value={customerFormData.state}
                    onChange={handleStateChange}
                    onBlur={() => markFieldTouched("state")}
                    className={getInputClass("state")}
                    disabled={
                      !customerFormData.city ||
                      isLoadingStates ||
                      availableStates.length === 0
                    }
                  >
                    <option value="">
                      {isLoadingStates
                        ? "Loading states..."
                        : !customerFormData.city
                          ? "First select city"
                          : "Select State"}
                    </option>
                    {availableStates.map((state) => (
                      <option key={state.id} value={state.stateName}>
                        {state.stateName}
                      </option>
                    ))}
                  </select>
                  {isLoadingStates && (
                    <div className="absolute right-3 top-3">
                      <svg
                        className="animate-spin h-5 w-5 text-blue-500"
                        viewBox="0 0 24 24"
                      >
                        <circle
                          className="opacity-25"
                          cx="12"
                          cy="12"
                          r="10"
                          stroke="currentColor"
                          strokeWidth="4"
                          fill="none"
                        />
                        <path
                          className="opacity-75"
                          fill="currentColor"
                          d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
                        />
                      </svg>
                    </div>
                  )}
                </div>
                {touchedFields.state && errors.state && (
                  <p className="text-sm text-red-600 mt-1">{errors.state}</p>
                )}
              </div>

              {/* Pincode */}
              <div>
                <label className="block font-medium text-gray-700 mb-1">
                  Pincode <span className="text-red-500">*</span>
                </label>
                <input
                  name="pincode"
                  type="text"
                  placeholder="6-digit pincode"
                  maxLength={6}
                  value={customerFormData.pincode}
                  onChange={handleFieldChange}
                  onBlur={() => markFieldTouched("pincode")}
                  className={getInputClass("pincode", false)}
                />
                {touchedFields.pincode && errors.pincode && (
                  <p className="text-sm text-red-600 mt-1">{errors.pincode}</p>
                )}
              </div>
            </div>
          </div>

          {/* Submit Button */}
          <div className="flex justify-center pt-4">
            <button
              type="submit"
              disabled={isLoading || isLoadingCities || isLoadingStates}
              className="px-10 py-3 text-white bg-blue-600 rounded-full hover:bg-blue-700 
                         disabled:opacity-50 disabled:cursor-not-allowed transition-all
                         font-semibold text-lg shadow-md hover:shadow-lg"
            >
              {isLoading ? (
                <span className="flex items-center gap-2">
                  <svg className="animate-spin h-5 w-5" viewBox="0 0 24 24">
                    <circle
                      className="opacity-25"
                      cx="12"
                      cy="12"
                      r="10"
                      stroke="currentColor"
                      strokeWidth="4"
                      fill="none"
                    />
                    <path
                      className="opacity-75"
                      fill="currentColor"
                      d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"
                    />
                  </svg>
                  Saving...
                </span>
              ) : (
                "Register Customer"
              )}
            </button>
          </div>
        </form>
      </div>
    </div>
  );
};

// Reusable Input
const Input = ({
  label,
  name,
  type = "text",
  icon,
  value,
  onChange,
  onBlur,
  error,
  touched,
  getInputClass,
  maxLength,
  required = false,
  placeholder,
}: any) => (
  <div>
    <label className="block font-medium text-gray-700 mb-1">
      {label} {required && <span className="text-red-500">*</span>}
    </label>
    <div className="relative">
      <input
        type={type}
        name={name}
        value={value}
        onChange={onChange}
        onBlur={() => onBlur(name)}
        className={getInputClass(name, true)}
        maxLength={maxLength}
        placeholder={placeholder}
      />
      <span className="absolute left-3 top-3 text-gray-400">{icon}</span>
    </div>
    {touched && error && <p className="text-sm text-red-600 mt-1">{error}</p>}
  </div>
);

// Reusable Select
const SelectField = ({
  label,
  name,
  icon,
  options,
  value,
  onChange,
  onBlur,
  getInputClass,
}: any) => (
  <div>
    <label className="block font-medium text-gray-700 mb-1">{label}</label>
    <div className="relative">
      <select
        name={name}
        value={value}
        onChange={onChange}
        onBlur={() => onBlur(name)}
        className={getInputClass(name, true)}
      >
        <option value="">Select {label}</option>
        {options.map((o: string) => (
          <option key={o} value={o}>
            {o}
          </option>
        ))}
      </select>
      <span className="absolute left-3 top-3 text-gray-400">{icon}</span>
    </div>
  </div>
);

export default CustomerForm;
